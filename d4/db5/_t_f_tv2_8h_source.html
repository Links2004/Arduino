153  void WRITE_Package(INT16U *data, INT8U howmany)
154  {
155  INT16U data1 = 0;
156  INT8U data2 = 0;
157 
158  TFT_DC_HIGH;
159  TFT_CS_LOW;
160  INT8U count=0;
161  for(count=0;count<howmany;count++)
162  {
163  data1 = data[count]>>8;
164  data2 = data[count]&0xff;
165  SPI.transfer(data1);
166  SPI.transfer(data2);
167  }
168  TFT_CS_HIGH;
169  }

171  INT8U Read_Register(INT8U Addr, INT8U xParameter)
172  {
173  INT8U data=0;

175  WRITE_DATA(0x10+xParameter); /* 0x11 is the first Parameter */
176  TFT_DC_LOW;

178  SPI.transfer(Addr);
179  TFT_DC_HIGH;
180  data = SPI.transfer(0);
181  TFT_CS_HIGH;
