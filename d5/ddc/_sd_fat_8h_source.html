106  return (fatDate >> 5) & 0XF;
107 }

110  return fatDate & 0X1F;
111 }

114  return hour << 11 | minute << 5 | second >> 1;
115 }

118  return fatTime >> 11;

122  return(fatTime >> 5) & 0X3F;
