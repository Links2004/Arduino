69  size_t read(char* dst, size_t size) {
70  size_t bytes_available = getSize();
71  size_t size_to_read = (size < bytes_available) ? size : bytes_available;
72  size_t size_read = size_to_read;
73  if(_end < _begin && size_to_read > (size_t)(_bufend - _begin)) {
74  size_t top_size = _bufend - _begin;
75  memcpy(dst, _begin, top_size);
76  _begin = _buf;
77  size_to_read -= top_size;
