155  size_t read(char* dst, size_t size) {
156  if(!_rx_buf) return 0;
157 
158  size_t max_size = _rx_buf->tot_len - _rx_buf_offset;
159  size = (size < max_size) ? size : max_size;
160 
161  DEBUGV(":rd %d, %d, %d\r\n", size, _rx_buf->tot_len, _rx_buf_offset);
162  size_t size_read = 0;
163  while(size) {
164  size_t buf_size = _rx_buf->len - _rx_buf_offset;
165  size_t copy_size = (size < buf_size) ? size : buf_size;
166  DEBUGV(":rdi %d, %d\r\n", buf_size, copy_size);
167  os_memcpy(dst, reinterpret_cast<char*>(_rx_buf->payload) + _rx_buf_offset, copy_size);
168  dst += copy_size;
169  _consume(copy_size);
170  size -= copy_size;
171  size_read += copy_size;
172  }
173  return size_read;
174  }

200  size_t write(const char* data, size_t size) {
201  if(!_pcb) {
202  DEBUGV(":wr !_pcb\r\n");
203  return 0;
204  }
205 
206  if(size == 0) {
207  return 0;
208  }
209 
210  size_t room = tcp_sndbuf(_pcb);
211  size_t will_send = (room < size) ? room : size;
212  err_t err = tcp_write(_pcb, data, will_send, 0);
213  if(err != ERR_OK) {
214  DEBUGV(":wr !ERR_OK\r\n");
215  return 0;
216  }
217 
218  _size_sent = will_send;
219  DEBUGV(":wr\r\n");
220  tcp_output( _pcb );
221  _send_waiting = true;
222  delay(5000); // max send timeout
223  _send_waiting = false;
224  DEBUGV(":ww\r\n");
225  return will_send - _size_sent;
226  }

237  void _consume(size_t size) {
238  ptrdiff_t left = _rx_buf->len - _rx_buf_offset - size;
239  if(left > 0) {
240  _rx_buf_offset += size;
241  } else if(!_rx_buf->next) {
242  DEBUGV(":c0 %d, %d\r\n", size, _rx_buf->tot_len);
243  if(_pcb) tcp_recved(_pcb, _rx_buf->len);
244  pbuf_free(_rx_buf);
245  _rx_buf = 0;
246  _rx_buf_offset = 0;
247  } else {
248  DEBUGV(":c %d, %d, %d\r\n", size, _rx_buf->len, _rx_buf->tot_len);
249  auto head = _rx_buf;
250  _rx_buf = _rx_buf->next;
251  _rx_buf_offset = 0;
252  pbuf_ref(_rx_buf);
253  if(_pcb) tcp_recved(_pcb, head->len);
254  pbuf_free(head);
255  }
256  }

258  err_t _recv(tcp_pcb* pcb, pbuf* pb, err_t err) {
259 
260  if(pb == 0) // connection closed
261  {
262  DEBUGV(":rcla\r\n");
263  return abort();
264  }
265 
266  if(_rx_buf) {
267  // there is some unread data
268  // chain the new pbuf to the existing one
269  DEBUGV(":rch %d, %d\r\n", _rx_buf->tot_len, pb->tot_len);
270  pbuf_cat(_rx_buf, pb);
271  } else {
272  DEBUGV(":rn %d\r\n", pb->tot_len);
273  _rx_buf = pb;
274  _rx_buf_offset = 0;
275  }
276  return ERR_OK;
277  }
