114 template <typename T>
115 inline size_t WiFiClient::write(T& source, size_t unitSize) {
116  std::unique_ptr<uint8_t[]> buffer(new uint8_t[unitSize]);
117  size_t size_sent = 0;
118  while(true) {
119  size_t left = source.available();
120  if (!left)
121  break;
122  size_t will_send = (left < unitSize) ? left : unitSize;
123  source.read(buffer.get(), will_send);
124  size_t cb = write(buffer.get(), will_send);
125  size_sent += cb;
126  if (cb != will_send) {
127  break;
128  }
129  }
130  return size_sent;
131 }
