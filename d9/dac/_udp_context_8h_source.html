233  size_t append(const char* data, size_t size)
234  {
235  if (!_tx_buf_head || _tx_buf_head->tot_len < _tx_buf_offset + size)
236  {
237  _reserve(_tx_buf_offset + size);
238  }
239 
240  size_t left_to_copy = size;
241  while(left_to_copy)
242  {
243  // size already used in current pbuf
244  size_t used_cur = _tx_buf_offset - (_tx_buf_head->tot_len - _tx_buf_cur->tot_len);
245  size_t free_cur = _tx_buf_cur->len - used_cur;
246  if (free_cur == 0)
247  {
248  _tx_buf_cur = _tx_buf_cur->next;
249  continue;
250  }
251  size_t will_copy = (left_to_copy < free_cur) ? left_to_copy : free_cur;
252  os_memcpy(reinterpret_cast<char*>(_tx_buf_cur->payload) + used_cur, data, will_copy);
253  _tx_buf_offset += will_copy;
254  left_to_copy -= will_copy;
255  data += will_copy;
256  }
257  return size;
258  }

260  void send(ip_addr_t* addr = 0, uint16_t port = 0)
261  {
262  size_t orig_size = _tx_buf_head->tot_len;
263 
264  size_t data_size = _tx_buf_offset;
265  size_t size_adjustment = orig_size - data_size;
266  for (pbuf* p = _tx_buf_head; p; p = p->next)
267  {
268  p->tot_len -= size_adjustment;
269  if (!p->next)
270  {
271  p->len = p->tot_len;
272  }
273  }
274 
275  if (!addr) {
276  addr = &_dest_addr;
277  port = _dest_port;
278  }
279 
280  uint16_t old_ttl = _pcb->ttl;
281  if (ip_addr_ismulticast(addr)) {
282  _pcb->ttl = _multicast_ttl;
283  }
284 
285  udp_sendto(_pcb, _tx_buf_head, addr, port);
286 
287  _pcb->ttl = old_ttl;
288 
289  for (pbuf* p = _tx_buf_head; p; p = p->next)
290  {
291  p->tot_len += size_adjustment;
292  if (!p->next)
293  {
294  p->len = p->tot_len;
295  }
296  }
297 
298  pbuf_free(_tx_buf_head);
299  _tx_buf_head = 0;
300  _tx_buf_cur = 0;
301  _tx_buf_offset = 0;
302  }

306  void _reserve(size_t size)
307  {
308  const size_t pbuf_unit_size = 512;
309  if (!_tx_buf_head)
310  {
311  _tx_buf_head = pbuf_alloc(PBUF_TRANSPORT, pbuf_unit_size, PBUF_RAM);
312  _tx_buf_cur = _tx_buf_head;
313  _tx_buf_offset = 0;
314  }
315 
316  size_t cur_size = _tx_buf_head->tot_len;
317  if (size < cur_size)
318  return;
319 
320  size_t grow_size = size - cur_size;
321 
322  while(grow_size)
323  {
324  pbuf* pb = pbuf_alloc(PBUF_TRANSPORT, pbuf_unit_size, PBUF_RAM);
325  pbuf_cat(_tx_buf_head, pb);
326  if (grow_size < pbuf_unit_size)
327  return;
328  grow_size -= pbuf_unit_size;
329  }
330  }
